# Este arquivo lista todas as bibliotecas Python necessárias para executar
# o sistema de roteamento de modelos LLM.

# Google GenAI SDK (novo SDK unificado)
# Substitui o google-cloud-aiplatform (descontinuado em Jun/2025, remoção Jun/2026)
# Usado para integração com os modelos Gemini (Pro e Flash) via Vertex AI
# Ref: https://cloud.google.com/vertex-ai/generative-ai/docs/deprecations/genai-vertexai-sdk
google-genai>=1.0.0

# Python Dotenv - Carregamento de variáveis de ambiente
# Usado para carregar configurações do arquivo .env
python-dotenv>=1.0.0

# Necessário para carregar as políticas de roteamento e configurações

# Jinja2 - Engine de templates
# Usado para processar o template audit_master.jinja2 e injetar variáveis

jinja2>=3.1.2

# Rich - Biblioteca para output formatado no terminal
# Usado no main.py para criar interfaces bonitas com tabelas, painéis e cores
# Melhora a experiência de visualização durante a demonstração
rich>=13.7.0

# Pydantic - Validação de dados com type hints
# Útil para validar estruturas de dados e garantir type safety
# Pode ser usado para validar respostas do LLM e configurações YAML
pydantic>=2.5.0

# Pytest - Framework de testes unitários
# Usado para criar e executar testes automatizados
# Garante qualidade e confiabilidade do código
pytest>=7.4.0
pytest-cov>=4.1.0  # Para cobertura de código

# Tiktoken - Contagem precisa de tokens para modelos Gemini
